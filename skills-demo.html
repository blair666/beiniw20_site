Data Processing Pipeline

import pandas as pd
import numpy as np
from sklearn.preprocessing import StandardScaler

def process_data(df):
    # Clean and transform data
    df_clean = df.dropna()
    
    # Feature engineering
    df_clean['experience_squared'] = df_clean['experience'] ** 2
    df_clean['edu_exp_interaction'] = (
        df_clean['education'] * df_clean['experience']
    )
    
    # Normalize features
    scaler = StandardScaler()
    features = ['experience', 'education', 
                'experience_squared', 'edu_exp_interaction']
    df_clean[features] = scaler.fit_transform(df_clean[features])
    
    return df_clean, scaler

ML Model Training

from sklearn.linear_model import LinearRegression
from sklearn.metrics import r2_score, mean_squared_error

class SalaryPredictor:
    def __init__(self):
        self.model = LinearRegression()
        self.scaler = None
        
    def train(self, X, y):
        self.model.fit(X, y)
        predictions = self.model.predict(X)
        
        return {
            'r2_score': r2_score(y, predictions),
            'rmse': np.sqrt(mean_squared_error(y, predictions))
        }
    
    def predict(self, experience, education):
        # Feature engineering
        features = np.array([[
            experience, education,
            experience**2, education*experience
        ]])
        
        return self.model.predict(features)[0]

Spring Boot REST Controller

                            
@RestController
@RequestMapping("/api/users")
@Validated
public class UserController {
    
    @Autowired
    private UserService userService;
    
    @GetMapping
    public ResponseEntity<List<User>> getAllUsers(
            @RequestParam(defaultValue = "0") int page,
            @RequestParam(defaultValue = "10") int size) {
        
        Pageable pageable = PageRequest.of(page, size);
        Page<User> users = userService.findAll(pageable);
        
        return ResponseEntity.ok()
            .header("X-Total-Count", String.valueOf(users.getTotalElements()))
            .body(users.getContent());
    }
    
    @PostMapping
    public ResponseEntity<User> createUser(
            @Valid @RequestBody CreateUserRequest request) {
        
        User user = userService.createUser(request);
        return ResponseEntity.status(HttpStatus.CREATED).body(user);
    }
}

                        

                        

                            
Service Layer with Validation

                            
@Service
@Transactional
public class UserService {
    
    @Autowired
    private UserRepository userRepository;
    
    @Autowired
    private EmailService emailService;
    
    public User createUser(CreateUserRequest request) {
        // Validate business rules
        if (userRepository.existsByEmail(request.getEmail())) {
            throw new BusinessException("Email already exists");
        }
        
        // Create user entity
        User user = User.builder()
            .name(request.getName())
            .email(request.getEmail())
            .age(request.getAge())
            .createdAt(LocalDateTime.now())
            .build();
        
        // Save and send welcome email
        User savedUser = userRepository.save(user);
        emailService.sendWelcomeEmail(savedUser);
        
        return savedUser;
    }
}
